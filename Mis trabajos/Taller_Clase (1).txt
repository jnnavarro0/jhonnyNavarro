||==Calidad de Servicios (Route ESMERALDA)==||

conf ter
class-map match-all ALTA
match ip dscp af11 
match protocol ospf 
class-map match-all MEDIA
match ip dscp af31
match protocol ssh
class-map match-all BAJA
match ip dscp af41
match protocol http
match protocol https
exit
policy-map ALERTAS
class ALTA
set precedence 6
class MEDIA
set precedence 4
class BAJA
set precedence 2
exit
exit
interface serial 0/1/0
service-policy output ALERTAS
